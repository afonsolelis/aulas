import io
import numpy as np
import pandas as pd
import numpy_financial as npf
from google.colab import files

print("1) Faça upload de um arquivo .xlsx ou .csv")
uploaded = files.upload()

nome_arquivo = list(uploaded.keys())[0]

# 2) Ler arquivo conforme extensao
if nome_arquivo.lower().endswith(".xlsx"):
    df = pd.read_excel(io.BytesIO(uploaded[nome_arquivo]))
    print(f"Arquivo Excel lido: {nome_arquivo}")
else:
    df = pd.read_csv(io.BytesIO(uploaded[nome_arquivo]))
    print(f"Arquivo CSV lido: {nome_arquivo}")

print("\n2) Primeiras linhas da base original")
display(df.head())

# 3) Padronizacao minima esperada
# Ajuste os nomes abaixo se sua planilha vier com nomes diferentes
mapa_colunas = {
    "Periodo": "periodo",
    "Período": "periodo",
    "Fluxo de Caixa": "fluxo_caixa",
    "Fluxo_Caixa": "fluxo_caixa",
    "fluxo": "fluxo_caixa"
}
df = df.rename(columns=mapa_colunas)

# 4) Validacoes basicas
colunas_obrigatorias = ["periodo", "fluxo_caixa"]
faltantes = [c for c in colunas_obrigatorias if c not in df.columns]
if faltantes:
    raise ValueError(f"Colunas obrigatorias ausentes: {faltantes}")

# 5) Limpeza
df["periodo"] = pd.to_datetime(df["periodo"], errors="coerce")

# Troca virgula decimal por ponto, se necessario
df["fluxo_caixa"] = (
    df["fluxo_caixa"]
    .astype(str)
    .str.replace(".", "", regex=False)
    .str.replace(",", ".", regex=False)
)
df["fluxo_caixa"] = pd.to_numeric(df["fluxo_caixa"], errors="coerce")

df = df.dropna(subset=["periodo", "fluxo_caixa"]).sort_values("periodo")

print("\n3) Base limpa")
display(df)
print(df.info())

# 6) Calculos financeiros
fluxos = df["fluxo_caixa"].to_list()
tma_mensal = 0.012  # 1,2% ao mes (ajuste conforme caso)

vpl = npf.npv(tma_mensal, fluxos)
tir = npf.irr(fluxos)

acumulado = np.cumsum(fluxos)
indices_positivos = np.where(acumulado >= 0)[0]
payback = int(indices_positivos[0]) if len(indices_positivos) > 0 else None

print("\n4) Resultado")
print(f"VPL (TMA = {tma_mensal:.2%}): R$ {vpl:,.2f}")
print(f"TIR: {tir:.2%}")
if payback is not None:
    print(f"Payback simples: {payback} periodos")
else:
    print("Payback simples: nao atingido no horizonte da base")

# 7) Exportar base tratada + acumulado para entrega
df_resultado = df.copy()
df_resultado["fluxo_acumulado"] = acumulado

saida = "resultado_analise.csv"
df_resultado.to_csv(saida, index=False)
files.download(saida)

print(f"\nArquivo gerado: {saida}")
